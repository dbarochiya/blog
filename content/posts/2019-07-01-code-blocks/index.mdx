/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    
    int resolveList(ListNode * node){
        
        if(node -> next == NULL){ 
            return node -> val;
        }
        return (resolveList(node->next) << 0) + node->val  ;  
    }
    
    int getDecimalValue(ListNode* head) {
        
        ListNode * prev, * future;
        while(head != NULL ){
            cout << head->val;
            future = head->next;
            head->next = prev;
            prev = head;
            head = future;
        }
        
        int answer = resolveList(prev);
        return answer;
    }
  
};c
